void setup() {
pinMode(LED_BUILTIN, OUTPUT);
pinMode(3, OUTPUT);
}

unsigned long interval = 5000;
bool flag = false;
int s = 0;

void loop() {
if (millis() > interval) {
interval += random(2000, 10000);
s = random(7);
flag = true;
}
switch (s) {
case 0: zero(flag); break;
case 1: one(); break;
case 2: two(); break;
case 3: three(); break;
case 4: four(); break;
case 5: five(flag); break;
case 6: six(); break;
case 7: seven(); break;
}

flag = false;
}

void zero(bool flag) {
if (flag) {
for (int i = 0; i < 10; i++) {
digitalWrite(3, HIGH);
delay(50);
digitalWrite(3, LOW);
delay(50);
}
}
digitalWrite(3, HIGH);
}

int brightness = 0;
int fadeAmount = 5;
void one() {
analogWrite(3, brightness);
brightness += fadeAmount;
if (brightness <= 0 || brightness >= 255) {
fadeAmount = -fadeAmount;
}
delay(30);
}

void two() {
digitalWrite(3, HIGH);
delay(500);
digitalWrite(3, LOW);
delay(500);
}

void three() {
digitalWrite(3, HIGH);
delay(900);
digitalWrite(3, LOW);
delay(100);
}

bool fourFlag = false;
void four() {
analogWrite(3, brightness);

if (fourFlag) {
brightness += random(10);
} else {
brightness -= random(10);
}

if (brightness >= 255) {
fourFlag = true;
} else if (brightness <= 0) {
fourFlag = false;
}
}

void five(bool flag) {
if (flag) {
for (int i = 0; i < 3; i++) {
digitalWrite(3, HIGH);
delay(500);
digitalWrite(3, LOW);
delay(500);
}
} else {
four();
}
}

void six() {
digitalWrite(3, HIGH);
delay(3000);
digitalWrite(3, LOW);
delay(3000);
}

void seven() {
digitalWrite(3, HIGH);
delay(3000);
digitalWrite(3, LOW);
delay(1000);
}
